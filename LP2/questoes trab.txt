    printf("A soma dos %d primeiros naturais impares é = %d\n",n,soma_impares(n));

    return 0;
}

int soma_impares(int n){
    int soma = 0,i;

    for(i = 0;i <= n;i++){
        if(i % 2 != 0){
        soma += i;
        }
    }
    return soma;
}

  scanf("%f", &raio);

  clac_esfera(raio, &area_sup, &vol_esf);

  printf("\nArea da superficie: %.3f\nVolume da esfera: %.3f\n",area_sup, vol_esf);

}

int clac_esfera(float r, float*area, float*volume){
  *area = (float)4*PI*pow(r,2);
  *volume = (float)(4*PI*pow(r,3))/3;
}

void inverte(int n, int *vet);

int main(){
    int n;

    printf("Informe o tamanho do vetor: \n");
    scanf("%d",&n);

    int vet[n],i;
    printf("Digite os valores do vetor:\n");
    for(i=0;i<n;i++){
        scanf("%d",&vet[i]);
    }

    inverte(n,vet);
    int j;
    for(j=0;j<n;j++){
    printf("%d ",vet[j]);
    }

    return 0;
}
void inverte(int n,int *vet) {
  int i,aux;

  for (i = 0; i < n/2; i++) {
    aux = vet[i];
    vet[i] = vet[n-1-i];
    vet[n-1-i] = aux;
  }
  return;
}
